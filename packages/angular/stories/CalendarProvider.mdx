# CalendarProvider

`CalendarProviderComponent` (`<bc-calendar-provider>`) is the required outer wrapper for every Big Calendar. It holds the active view, the focus date, the event list, and all settings. Every component or inject composable inside reads from that shared state automatically via Angular's `InjectionToken`.

```ts
// app.component.ts
import { Component } from '@angular/core'
import { CalendarProviderComponent, CalendarComponent } from '@big-calendar/angular'
import { createTemporalLocalizer } from '@big-calendar/localizer-temporal'

const localizer = await createTemporalLocalizer({ locale: 'en-US', timeZone: 'UTC' })

@Component({
  standalone: true,
  imports: [CalendarProviderComponent, CalendarComponent],
  template: `
    <bc-calendar-provider [localizer]="loc" [events]="events" defaultDate="2026-06-15">
      <bc-calendar />
    </bc-calendar-provider>
  `,
})
export class AppComponent {
  readonly loc = localizer
  readonly events = [
    { id: '1', title: 'Team standup', start: '2026-06-15T09:00:00Z', end: '2026-06-15T09:30:00Z' }
  ]
}
```

### Uncontrolled (calendar manages its own state)

<table>
<thead><tr><th>Input</th><th>Type</th><th>Default</th><th>Description</th></tr></thead>
<tbody>
<tr>

<td><code>ViewKey</code></td>
<td><code>'month'</code></td>
<td>Which view the calendar opens on: <code>'month'</code>, <code>'week'</code>, <code>'work_week'</code>, <code>'day'</code>, <code>'agenda'</code>.</td>
</tr>
<tr>
<td><code>defaultDate</code></td>
<td><code>string</code></td>
<td>Today</td>
<td>The date the calendar focuses on when it first loads, as an ISO 8601 string (e.g. <code>"2026-06-15"</code>).</td>
</tr>

### Controlled (your state drives the calendar)

```ts
@Component({
  standalone: true,
  imports: [CalendarProviderComponent, CalendarComponent],
  template: `
    <bc-calendar-provider
      [localizer]="loc"
      [events]="events"
      [view]="view()"
      [date]="date()"
      [onView]="handleView"
      [onNavigate]="handleNavigate"
    >
      <bc-calendar />
    </bc-calendar-provider>
  `,
})
export class AppComponent {
  readonly view = signal<ViewKey>('week')
  readonly date = signal('2026-06-15')
  readonly handleView = ({ view }: { view: ViewKey }) => this.view.set(view)
  readonly handleNavigate = ({ date }: { date: string }) => this.date.set(date)
}
```

## Accessors

By default Big Calendar reads `title`, `start`, `end`, `allDay`, `id`, `resourceId`, `type`, `draggable`, and `resizable` from your event objects. Override with `[accessors]` to match your data shape:

```ts
// Your events use 'name' and 'startsAt' / 'endsAt':
readonly accessors = {
  title: 'name',
  start: (event: MyEvent) => event.startsAt,
  end:   (event: MyEvent) => event.endsAt,
}
```

```html
<bc-calendar-provider [accessors]="accessors" ...>
```
