# Custom Components

Big Calendar lets you replace any visual piece of the calendar — event blocks, day headings, gutter labels, the show-more button, and more — with your own Angular templates.

In Angular, custom slot components are `ng-template` inputs on the individual view components (`<bc-month-view>`, `<bc-time-grid-view>`, `<bc-agenda-view>`). Each template receives its context object via `$implicit`.

## Using `<bc-calendar>` vs. individual view components

`<bc-calendar>` auto-dispatches to the correct view component based on the active view, but does not expose slot template inputs directly. To pass custom slot templates, use the individual view component alongside `<bc-calendar-provider>`:

```html
<!-- Instead of: <bc-calendar /> -->
<bc-month-view [bcMonthEvent]="myEventTemplate" />
```

You can use multiple view components inside one provider and show/hide them with `@if`:

```html
<bc-calendar-provider [localizer]="loc" [events]="events" [defaultView]="view()">
  @if (view() === 'month') {
    <bc-month-view [bcMonthEvent]="customEvent" />
  } @else {
    <bc-time-grid-view [bcTimeEvent]="customEvent" />
  }
</bc-calendar-provider>
```

## Minimal example

```ts
import { Component, TemplateRef, ViewChild } from '@angular/core'
import { CalendarProviderComponent, MonthViewComponent } from '@big-calendar/angular'

@Component({
  standalone: true,
  imports: [CalendarProviderComponent, MonthViewComponent],
  template: `
    <bc-calendar-provider [localizer]="loc" [events]="events" defaultView="month">
      <bc-month-view [bcMonthEvent]="boldEvent" />
    </bc-calendar-provider>
    <ng-template #boldEvent let-ctx>
      <strong>{{ ctx.title }}</strong>
    </ng-template>
  `,
})
export class CalendarWithCustomEventsComponent {
  @ViewChild('boldEvent') boldEvent!: TemplateRef<unknown>
  // ...
}
```

> **Note:** The `@ViewChild` decorator is required because Angular evaluates the `[bcMonthEvent]` binding after the view initialises. If you use a signal-based approach or the new `viewChild()` query, the timing is handled automatically.
